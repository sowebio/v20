43        procedure Backup_File (File_To_Backup : VString) is
44           Backup_File : constant VString := File_To_Backup & ".bak.";
45           Backup_To_Delete : Integer := 0;
46        begin 
47           for I in 0 .. 9 loop
48              if not Exists (Backup_File & To_VString (I)) then
49                 Move_File (File_To_Backup, Backup_File & To_VString (I));
50                 if I = 9 then
51                    Backup_To_Delete := 0;
52                 else
53                    Backup_To_Delete := I + 1;
54                 end if;
55                 if Exists (Backup_File & To_VString (Backup_To_Delete)) then
56                    Delete_File (Backup_File & To_VString (Backup_To_Delete));
57                 end if;
58                 exit;
59              end if; 
60           end loop;
61        end Backup_File;

69        procedure Copy_File (Source_Name, Target_Name : String) is
70           Destination_Valid : Boolean := True;
71        begin
72           -- Check if destination is current or contains a leading directory
73           if (Index (To_VString (Target_Name), "/") > 0) then
74              if not (Exists (Slice (To_VString (Source_Name), 1,
75                                  Index_Backward (To_VString (Source_Name), "/"))))
76              then
77                 Destination_Valid := False;
78                 Log.Err ("v20.Fls.Copy_File > Destination directory does not exist: " & Target_Name);
79              end if;
80           end if;  
81           if Destination_Valid then      
82              AD.Copy_File (Source_Name, Target_Name, Copy_Form);
83           end if;
84        exception
85           when Error : Ada.IO_Exceptions.Use_Error =>
86              Log_Err (Error);
87           when Error : Ada.IO_Exceptions.Name_Error =>
88              Log_Err (Error);
89        end Copy_File;

130        function Delete_Directory_Tree (Dir_Tree : String) return Boolean is
131           Result : Boolean := False;
132        begin
133           if not If_Root_Directory (Dir_Tree) then
134              if AD.Exists (Dir_Tree) then
135                 AD.Delete_Tree (Dir_Tree);
136                 if not AD.Exists (Dir_Tree) then
137                    Result := True;
138                 end if;
139              else
140                 Result := True;
141              end if;
142           else
143              Log.Err ("Fls.Delete_Directory_Tree - Attempt to delete a root directory: " & Dir_Tree);
144           end if;   
145           return Result;
146        exception
147           when Error : Ada.IO_Exceptions.Use_Error =>
148              Log_Err (Error);
149              return False;
150           when Error : Ada.IO_Exceptions.Name_Error =>
151              Log_Err (Error);
152              return False;
153        end Delete_Directory_Tree;

193        procedure Delete_Lines (File_Name, Pattern : String) is
194           File_Write : constant String := File_Name  & ".new";
195           Handle_Read, Handle_Write : Tio.File;
196           Line_Deleted : Boolean := False;
197           Line_Buffer : VString := +"";
198        begin
199           if Fls.Exists (File_Name) then
200              Tio.Open_Read (Handle_Read, File_Name);
201              if Tio.Is_Open (Handle_Read) then
202                 Tio.Create (Handle_Write, File_Write);
203                 if Tio.Is_Open (Handle_Write) then
204                    while not Tio.End_Of_File (Handle_Read) loop
205                       Tio.Get_Line (Handle_Read, Line_Buffer);
206                       if Index (Line_Buffer, Pattern) > 0 then
207                          Line_Deleted := True;
208                       else
209                          Tio.Put_Line (Handle_Write, Line_Buffer);
210                       end if;
211                    end loop;
212                    Tio.Close (Handle_Write);
213                 end if;
214                 Tio.Close (Handle_Read);
215              end if;
216              if Line_Deleted then
217                 Rename (File_Name, File_Name & ".bak");
218                 Rename (File_Write, File_Name);
219              end if;
220           else
221              Log.Err ("v20.Fls.Delete_Lines > Can't find: " & File_Name);
222           end if;
223        end Delete_Lines;

241        function Download_File (Url : VString;
242                                Dlfile : VString;
243                                Name : VString := +"";
244                                DlSize : Integer := 0) return Boolean is
245           Exec_Error : Integer;
246           Result : Boolean := False;
247           Message_Name : VString := Name;
248        begin
249           if Vst.Length (Name) = 0 then
250              Message_Name := Url;
251           end if;
252           
253           --  Processing existing file
254           if Fls.Exists (Dlfile) then
255              if DlSize > 0 then
256                 --  If Size correct
257                 if Fls.File_Size (Dlfile) = DlSize then
258                    Result := True;
259                    Log.Msg ("Keep existing and valid file: " & Dlfile);
260                 end if;
261              end if;
262              if not Result then
263                 Log.Msg ("Delete old file: " & Dlfile);
264                 Fls.Delete_File (Dlfile);
265              end if;
266           end if;
267           --  Proceed to download if needed
268           if not Result then
269              Log.Msg ("Download file: " & Message_Name);
270              
271              -- http1.1 to avoid curl error 'HTTP/2 stream 0 was not closed cleanly'
272              Sys.Shell_Execute ("curl --http1.1 --location --output " &
273                              Dlfile & " " & Url, Exec_Error);
274              if (Exec_Error = 0) and Fls.Exists (Dlfile) then
275                 if DlSize > 0 then
276                    if Fls.File_Size (Dlfile) = DlSize then
277                       Result := True;
278                    end if;
279                 else
280                    Result := True;
281                 end if;
282              end if;
283              if not Result then
284                 Log.Err ("Fls.Download_File > Download file failed: " & Message_Name);
285              end if;
286           end if;
287           return Result;
288        end Download_File;

297        function Extract_Directory (Name : VString) return VString is
298           Slash : constant VString := +"/";
299           Dir_End : Natural;
300           Dir_Result : VString := +"";
301        begin
302           Dir_End := Index_Backward (Name, Slash);
303           if (Dir_End > 1) then 
304              Dir_Result := Slice (Name, 1, Dir_End - 1);
305           end if;
306           return Dir_Result;
307        end Extract_Directory;

350        function If_Root_Directory (Dir_Tree : VString) return Boolean is
351           Test_Dir_Tree : VString := Dir_Tree;
352           Slash : constant VString := +"/";
353           Root_Dirs : constant VString := +"bin,boot,dev,etc,home,lib,lib32,lib64," & 
354              "libx32,lost+found,media,mnt,opt,proc,root,run,sbin,srv,sys," & 
355              "tmp,usr,var,Test_Delete_Directory_Tree";
356           Result : Boolean := False;
357        begin
358           -- Dir_Tree must be fully qualified, ie starting with a slash (/)
359           if Starts_With (Dir_Tree, Slash) then
360              --  Add a slash if Dir_Tree does not ends with it
361              if not Ends_With (Dir_Tree, Slash) then
362                 Test_Dir_Tree := Test_Dir_Tree & Slash;
363              end if;
364              --  If Dir_Tree counts only two slashes and at least one char between
365              if (Char_Count (Test_Dir_Tree, Slash) = 2) and
366                 (Length (Dir_Tree) > 2) then
367                 Test_Dir_Tree := Slice (Test_Dir_Tree, 2, Length (Test_Dir_Tree) - 1);
368                 if not Empty (Field_By_Name (Root_Dirs, Test_Dir_Tree, To_String (Slash))) then
369                    Result := True;
370                 end if;
371              end if;
372           end if;
373           return Result;
374        end If_Root_Directory;

404        procedure Rename (Old_Name, New_Name : String) is
405           use Ada.Directories;  --  For File_Kind.Directory visibility
406        begin
407           if Exists (New_Name) then
408              if not (Kind (New_Name) = Directory) then
409                 AD.Delete_File (New_Name);
410              end if;
411           end if;
412           AD.Rename (Old_Name, New_Name);
413        exception
414           when Error : Ada.IO_Exceptions.Use_Error =>
415              Log_Err (Error);
416           when Error : Ada.IO_Exceptions.Name_Error =>
417              Log_Err (Error);
418        end Rename;

436        function Search_Lines (File_Name, Pattern : String) return Boolean is
437           Handle_Read : Tio.File;
438           Line_Found : Boolean := False;
439           Line_Buffer : VString := +"";
440        begin
441           Tio.Open_Read (Handle_Read, File_Name);
442           if Tio.Is_Open (Handle_Read) then
443              while not Tio.End_Of_File (Handle_Read) loop
444                 Tio.Get_Line (Handle_Read, Line_Buffer);
445                 if Index (Line_Buffer, Pattern) > 0 then
446                    Line_Found := True;
447                    exit;
448                 end if;
449              end loop;
450              Tio.Close (Handle_Read);
451           end if;
452           return Line_Found;
453        end Search_Lines;
