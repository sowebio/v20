74     -- Bind -- Set a parameter of statement
75     --
76     --    Command   - The prepared command
77     --    Parameter - The position of
78     --    Value     - To be bound when missing then null is bound
79     --
80     -- The parameters to be bound are usually specified as ? in the  command
81     -- text  (see  Prepare). Each such parameter has to be bound to a value.
82     -- The position of a parameter is specified by its index,  i.e.  by  the
83     -- position  of  ?  in  the  command  text.  The first parameter has the
84     -- position 1. The variant with Value of access String type is used when
85     -- the  caller  takes  the  responsibility not to destroy string all the
86     -- time  the binding is used. In this case an attempt is made to prevent
87     -- extra copying of the string contents.
88     --
89     -- Exceptions :
90     --
91     --    Constraint_Error - Command or Parameter is invalid
92     --    Data_Error       - Data base error
93     --    End_Error        - Not found (table does not exist)
94     --    Status_Error     - Access errors
95     --    Use_Error        - File access related errors
96     --
97     
98        procedure Bind (Command : Statement; Parameter : Positive; Value : double);
99        procedure Bind (Command : Statement; Parameter : Positive; Value : int);
100        procedure Bind (Command : Statement; Parameter : Positive; Value : Integer_64);
101        procedure Bind (Command : Statement; Parameter : Positive; Value : String);
102        procedure Bind (Command : Statement; Parameter : Positive; Value : Stream_Element_Array);
103        procedure Bind (Command : Statement; Parameter : Positive; Value : access String);
104        procedure Bind (Command : Statement; Parameter : Positive; Value : access Stream_Element_Array);
105        procedure Bind (Command : Statement; Parameter : Positive; Value : chars_ptr; Length : size_t);
106        procedure Bind (Command : Statement; Parameter : Positive);

314     ------------------------------------------------------------------------
315     -- Step -- Execue prepared command
316     --
317     --    Command   - The statement being execured
318     --    Completed - The execution state
319     --
320     -- When  the  result is False, the Command execution has been completed.
321     -- In  this  case the next operation should be Reset. When the result is
322     -- True  there  is  a  row  of  data  produced  by the command. The next
323     -- operation can be Step to get another row or else Reset to  reset  the
324     -- statement. The statement life-cycle looks like this:
325     --
326     --    declare
327     --       Command : Statement := Prepare (DB, "SQL command");
328     --    begin
329     --       Bind (Command, ...); -- Binding parameters
330     --       while Step (Command) loop
331     --          ... Column (Command) ... -- Taking the results out
332     --       end loop;
333     --       Reset (Command);
334     --
335     -- After Reset the parameters can be rebound before another execution of
336     -- the parameter is initiated by doing Step.
337     --
338     -- Returns :
339     --
340     --    False when the execution was completed. The next
341     --
342     -- Exceptions :
343     --
344     --    Constraint_Error - Command is an invalid handle
345     --    Data_Error       - Data base error
346     --    End_Error        - Not found (table does not exist)
347     --    Status_Error     - Access error
348     --    Use_Error        - File open error
349     --
350     
351        procedure Step (Command : Statement);
352        function Step (Command : Statement) return Boolean;
