33        function HAC_Image (I : Integer) return String is
34           Im : constant String := Integer'Image (I);
35        begin
36           if I < 0 then
37              return Im;
38           else
39              return Im (Im'First + 1 .. Im'Last);
40           end if;
41        end HAC_Image;
42     
43        function HAC_Generic_Image (I : Abstract_Integer) return String is
44           Im : constant String := Abstract_Integer'Image (I);
45        begin
46           if I < 0 then
47              return Im;
48           else
49              return Im (Im'First + 1 .. Im'Last);  --  Remove the leading ' '.
50           end if;
51        end HAC_Generic_Image;

99        function Trim_Slashes (Source : VString) return VString is
100           Src : VString := Source;
101        begin
102           --  Specific case
103           if Src = "/" then
104              Src := +"";
105           end if;
106           --  Suppress leading slash(es)
107           while Index (Src, "/") = 1 and Length (Src) > 1 loop
108              Src := Slice (Src, 2, Length (Src));
109           end loop;
110           --  Suppress trailing slash(es)
111           while Index_Backward (Src, "/") = Length (Src) and Length (Src) > 1 loop
112              Src := Slice (Src, 1, Length (Src) - 1);
113           end loop;
114           return Src;
115        end Trim_Slashes;

284        function Tail_After_Match (Source : VString; Pattern : VString)
285                                   return VString is
286           Result : VString := +"";
287           Source_Length : constant Natural := Length (Source);
288           Pattern_Length : constant Natural := Length (Pattern);
289        begin
290           for I in reverse 1 .. Source_Length loop
291              if I <= Source_Length - Pattern_Length then
292                 if Slice (Source, I, I + Pattern_Length - 1) = Pattern then
293                    Result := Slice (Source, I + Pattern_Length, Source_Length);
294                    exit;
295                 end if;
296              end if;
297           end loop;
298           return Result;
299        end Tail_After_Match;
