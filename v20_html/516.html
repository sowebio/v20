111        procedure Check (Connection : SQLite_Handle; Result : int) is
112           function Message return String is
113     
114              -- 20211214 - sr
115              -- Avoid dependency with Strings_Edit.Integers and related packages
116              -- Code : constant String := '[' & Image (Integer (Result)) & ']';
117              Code : constant String := '[' & Interfaces.C.int'Image (Result) & ']';
118     
119           begin
120              if Connection = null then
121                 return Message (Result) & ' ' & Code;
122              else
123                 declare
124                    function Internal (Error : int)
125                       return chars_ptr;
126                    pragma Import (C, Internal, "sqlite3_errstr");
127                    Text : constant chars_ptr := Internal (Result);
128                 begin
129                    if Text = Null_Ptr then
130                       return Message (Result) & ' ' & Code;
131                    else
132                       return Value (Text) & ' ' & Code;
133                    end if;
134                 end;
135              end if;
136           end Message;
137        begin
138           case Result is
139              when SQLITE_OK =>
140                 return;
141              when SQLITE_PERM | SQLITE_CANTOPEN =>
142                 Raise_Exception (Use_Error'Identity, Message);
143              when SQLITE_ABORT    | SQLITE_BUSY | SQLITE_LOCKED |
144                   SQLITE_READONLY | SQLITE_AUTH =>
145                 Raise_Exception (Status_Error'Identity, Message);
146              when SQLITE_NOTFOUND | SQLITE_ERROR =>
147                 Raise_Exception (End_Error'Identity, Message);
148              when SQLITE_RANGE =>
149                 Raise_Exception (Constraint_Error'Identity, Message);
150              when others =>
151                 Raise_Exception (Data_Error'Identity, Message);
152           end case;
153        end Check;

440        function Column
441                 (  Command  : Statement;
442                    Position : Positive
443                 )  return String is
444           function Bytes
445                    (  Statement : SQLite_Handle;
446                       Index     : int
447                    )  return int;
448           pragma Import (C, Bytes, "sqlite3_column_bytes");
449           function Text
450                    (  Statement : SQLite_Handle;
451                       Index     : int
452                    )  return chars_ptr;
453           pragma Import (C, Text, "sqlite3_column_text");
454           Handle : constant SQLite_Handle := Ptr (Command.Handle).Handle;
455           Index  : constant int       := int (Position) - 1;
456           Result : constant chars_ptr := Text (Handle, Index);
457           Count  : constant int       := Bytes (Handle, Index);
458        begin
459           if Count = 0 then
460              return "";
461           else
462              return Value (Result, size_t (Count));
463           end if;
464        end Column;

466        function Column
467                 (  Command  : Statement;
468                    Position : Positive
469                 )  return Stream_Element_Array is
470           pragma Assert ((Stream_Element'Size) mod 8 = 0);
471           function Bytes
472                    (  Statement : SQLite_Handle;
473                       Index     : int
474                    )  return int;
475           pragma Import (C, Bytes, "sqlite3_column_bytes");
476           Handle : constant SQLite_Handle := Ptr (Command.Handle).Handle;
477           Index  : constant int := int (Position) - 1;
478           Count  : constant int := Bytes (Handle, Index);
479        begin
480           if Count = 0 then
481              return (1..0 => Stream_Element'First);
482           else
483              declare
484                 function Internal
485                          (  Statement : SQLite_Handle;
486                             Index     : int
487                          )  return Address;
488                 pragma Import (C, Internal, "sqlite3_column_blob");
489                 subtype Result_Array is
490                         Stream_Element_Array
491                         (  1
492                         .. Stream_Element_Offset
493                            (  (Count * 8) / Stream_Element'Size
494                         )  );
495                 Result : Result_Array;
496                 pragma Import (Ada, Result);
497                 for Result'Address use Internal (Handle, Index);
498              begin
499                 return Result;
500              end;
501           end if;
502        end Column;

569        procedure Finalize (Object : in out Data_Base_Object) is
570           function Internal (sqlite3 : SQLite_Handle) return int;
571           pragma Import (C, Internal, "sqlite3_close");
572        begin
573           if Object.User_Data /= null then
574              Release (Object.User_Data);
575           end if;
576           if Object.Handle /= null then
577              Check (Object.Handle, Internal (Object.Handle));
578              Object.Handle := null;
579           end if;
580           Finalize (Entity (Object));
581        exception
582           when others =>
583              Object.Handle := null;
584              Finalize (Entity (Object));
585        end Finalize;

628        function Init
629                 (  Destination      : Data_Base'Class;
630                    Destination_Name : String := "main";
631                    Source           : Data_Base'Class;
632                    Source_Name      : String := "main"
633                 )  return Backup is
634           function Internal
635                    (  Destination      : SQLite_Handle;
636                       Destination_Name : char_array;
637                       Source           : SQLite_Handle;
638                       Source_Source    : char_array
639                    )  return SQLite_Backup_Handle;
640           pragma Import (C, Internal, "sqlite3_backup_init");
641           From : constant SQLite_Handle := Ptr (Source.Handle).Handle;
642           To   : constant SQLite_Handle := Ptr (Destination.Handle).Handle;
643           This : constant Backup_Handles.Handle := Ref (new Backup_Object);
644        begin
645           Ptr (This).Handle :=
646              Internal
647              (  To,
648                 To_C (Destination_Name),
649                 From,
650                 To_C (Source_Name)
651              );
652           if Ptr (This).Handle = null then
653              Check (To, Errcode (To));
654           end if;
655           return (Handle => This);
656        end Init;

684        function Message (Result : int) return String is
685        begin
686           case Result is
687              when SQLITE_OK =>
688                 return "Success";
689              when SQLITE_ERROR =>
690                 return "SQL error or missing database";
691              when SQLITE_INTERNAL =>
692                 return "Internal logic error in SQLite";
693              when SQLITE_PERM =>
694                 return "Access permission denied";

860        function Trace_Callbak
861                 (  T : unsigned;
862                    C : Address;
863                    P : Address;
864                    X : Address
865                 )  return int is
866           package As_Integer is
867              new System.Address_To_Access_Conversions (Integer_64);
868           package As_Statement is
869              new System.Address_To_Access_Conversions (SQLite_Object);
870           function "+" is new Ada.Unchecked_Conversion (Address, chars_ptr);
871           This    : constant Data_Base_Object_Ptr := From_Address (C);
872           Base    : Data_Base;
873           Command : Statement;
874           procedure Create_Statement is
875           begin
876              Set (Command.Handle, new Immutable_Statement_Object);
877              declare
878                 Object : Statement_Object'Class renames
879                          Ptr (Command.Handle).all;
880              begin
881                 Object.Base := Base.Handle;
882                 Object.Handle :=
883                    As_Statement.To_Pointer (P).all'Unchecked_Access;
884              end;
885           end Create_Statement;
886        begin
887           if This.Use_Count = 0 then
888              return 0;
889           end if;
890           begin
891              case T is
892                 when SQLITE_TRACE_STMT =>
893                    if This.Do_Statement /= null then
894                       Set (Base.Handle, This);
895                       declare
896                          Query : constant String := Value (+X);
897                       begin
898                          Create_Statement;
899                          This.Do_Statement (Command, Query, This.User_Data);
900                       end;
901                    end if;
902                 when SQLITE_TRACE_PROFILE =>
903                    if This.Do_Profile /= null then
904                       Set (Base.Handle, This);
905                       Create_Statement;
906                       This.Do_Profile
907                       (  Command,
908                          Duration
909                          (  Long_Float (As_Integer.To_Pointer (X).all)
910                          *  0.000_000_001
911                          ),
912                          This.User_Data
913                       );
914                    end if;
915                 when SQLITE_TRACE_ROW =>
916                    if This.Do_Row /= null then
917                       Set (Base.Handle, This);
918                       Create_Statement;
919                       This.Do_Row (Command, This.User_Data);
920                    end if;
921                 when SQLITE_TRACE_CLOSE =>
922                    if This.Do_Close /= null then
923                       Set (Base.Handle, This);
924                       This.Do_Close (This.User_Data);
925                    end if;
926                 when others =>
927                    null;
928              end case;
929           exception
930              when others =>
931                 null;
932           end;
933           return 0;
934        end Trace_Callbak;

958        procedure Set_Trace (Base : in out Data_Base_Object'Class) is
959           function Internal
960                    (  db       : SQLite_Handle;
961                       Mask     : unsigned;
962                       Callback : Trace_Callbak_Ptr;
963                       Context  : Address
964                    )  return int;
965           pragma Import (C, Internal, "sqlite3_trace_v2");
966           Mask : unsigned := 0;
967        begin
968           if (  Base.Do_Statement = null
969              and then
970                 Base.Do_Profile = null
971              and then
972                 Base.Do_Row = null
973              and then
974                 Base.Do_Close = null
975              )  then
976              Check
977              (  Base.Handle,
978                 Internal
979                 (  Base.Handle,
980                    0,
981                    null,
982                    Null_Address
983              )  );
984           else
985              if Base.Do_Statement /= null then
986                 Mask := Mask or SQLITE_TRACE_STMT;
987              end if;
988              if Base.Do_Profile /= null then
989                 Mask := Mask or SQLITE_TRACE_PROFILE;
990              end if;

992                 Mask := Mask or SQLITE_TRACE_ROW;
993              end if;
994              if Base.Do_Close /= null then
995                 Mask := Mask or SQLITE_TRACE_CLOSE;
996              end if;
997              Check
998              (  Base.Handle,
999                 Internal
1000                 (  Base.Handle,
1001                    Mask,
1002                    Trace_Callbak'Access,
1003                    Base'Address
1004              )  );
1005           end if;
1006        end Set_Trace;
