28     -- The  package Object provides the type Entity serving as the base type
29     -- for all derived types which require a garbage collection and  handles
30     -- (see the child package Object.Handle). The  type  Entity  is  derived
31     -- from Ada.Finalization.Limited_Controlled.
32     --
33     with Ada.Finalization;
34     
35     package Object is
36     --
37     -- Entity -- The abstract base of all reference counted objects.  Though
38     --           the  component  Use_Count  is  exposed,  it  shall never be
39     --           updated directly.
40     --
41        type Entity is new Ada.Finalization.Limited_Controlled with record
42           Use_Count : Natural := 0; -- Reference count
43        end record;
44        type Entity_Ptr is access all Entity'Class;
45     --
46     -- Decrement_Count -- Decrement object's use count
47     --
48     --    Object    - The object
49     --    Use_Count - The new use count value
50     --
51     -- This procedure can be  overridden  in  order  to  provide  additional
52     -- semantics of decrementing object reference count.
53     --
54     -- Exceptions :
55     --
56     --    Program_Error - Use count is zero
57     --
58        procedure Decrement_Count
59                  (  Object    : in out Entity;
60                     Use_Count : out Natural
61                  );
62     --
63     -- Equal, Less -- Comparisons
64     --
65     --    Left  - An object (dispatching)
66     --    Right - An object (class)
67     --    Flag  - Recursion flag (used by implementation)
68     --
69     -- Note  that  because  Ada  95  has  no  proper  multiple dispatch, one
70     -- argument has to be made class-wide to avoid exceptions when type tags
71     -- differ.  When  the  second  argument  is  in  Handle'Class  then  the
72     -- addresses of of the objects are  compared.  Otherwise,  the  function
73     -- re-dispatches  according  to  the second argument. A derived type may
74     -- override this method in the first argument, which  action  will  then
75     -- have a symmeric effect. The parameter  Flag  is  used  internally  to
76     -- distinguish recursive calls. An implementation sets this parameter to
77     -- true when it makes a recursive call the function. It is necessary  to
78     -- avoid infinite recursion when the function is inherited.
79     --
80     -- Returns :
81     --
82     --    The result of comparison
83     --
84        function Equal
85                 (  Left  : Entity;
86                    Right : Entity'Class;
87                    Flag  : Boolean := False
88                 )  return Boolean;
89        function Less
90                 (  Left  : Entity;
91                    Right : Entity'Class;
92                    Flag  : Boolean := False
93                 )  return Boolean;
94     --
95     -- Finalize -- To be called by any derived type
96     --
97     --    Object - The object to finalized
98     --
99     -- Exceptions :
100     --
101     --    Program_Error - The object is still in use
102     --
103        procedure Finalize (Object : in out Entity);
104     --
105     -- Increment_Count -- Increment object's use count
106     --
107     --    Object - The object
108     --
109     -- This procedure can be  overridden  in  order  to  provide  additional
110     -- semantics of incrementing the object reference count. Typically it is
111     -- used in order to make reference counting tasking safe.
112     --
113        procedure Increment_Count (Object : in out Entity);
114     --
115     -- Initialize -- To be called by any derived type
116     --
117     --    This - The object to initialize
118     --
119        procedure Initialize (Object : in out Entity);
120     --
121     -- Release -- Decrement object's use count
122     --
123     --    Ptr - To the object
124     --
125     -- The object pointed by Ptr is deleted if its use count in 1. Otherwise
126     -- the use count is decremented. Nothing happens if Ptr is null.
127     --
128     -- Exceptions :
129     --
130     --    Program_Error - Use count is already zero
131     --
132        procedure Release (Ptr : in out Entity_Ptr);
133     
134     ------------------------------------------------------------------------
135     -- Object use traceback. The default implementation of these  procedures
136     -- do  nothing.  There is an implementation that support tracebug is the
137     -- corresponding project option is chosen.
138     --
139     -- Put_Traceback -- Write traceback of an object
140     --
141     --    Object - To trace
142     --
143        procedure Put_Traceback (Object : Entity'Class);
144     --
145     -- Set_Trace_File -- Set trace file, the default is standard output
146     --
147     --    File - The file name to open
148     --
149     -- Exceptions :
150     --
151     --    Upon file creation
152     --
153        procedure Set_Trace_File (File : String);
154     
155     private
156        pragma Inline (Decrement_Count);
157        pragma Inline (Increment_Count);
158     
159     end Object;
