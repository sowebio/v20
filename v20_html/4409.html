38        function Get_Password return VString is
39           Exit_Char : constant Character := Character'Val(10); -- LF (return code key)
40           Current_Char : Character;
41           Result : VString := +"";
42        begin
43           Tio.Put ("Password:");
44           while True loop
45              Get_Immediate (Current_Char);
46              --Put (To_Val(To_VString(Current_Char)));
47              if Current_Char = Exit_Char then
48                 exit;
49              else
50                 Result := Result & To_VString (Current_Char);
51              end if;
52           end loop;
53           Tio.Line;
54           return Result;
55        end Get_Password;

200        procedure Open_Conf (Handle : in out File; Name : String ;
201                            Wipe_Before_Process : Boolean := False ;
202                            Permissions : VString := +"") is
203           Dir_File : constant VString := Fls.Extract_Directory (Name);
204           SE_Result : Integer := 0;
205        begin
206           --  Preventive house keeping
207           if Fls.Exists (Name) then
208              Fls.Backup_File (Name);
209              if Wipe_Before_Process then
210                 Fls.Delete_File (Name);
211              end if;
212           end if;
213     
214           if Fls.Exists (Name) then
215              Append (Handle, Name);
216           else
217              -- Ensure that the complete tree structure exists before creating file
218              if Fls.Create_Directory_Tree (Dir_File) then
219                 Create (Handle, Name);
220              else
221                 Log.Err ("Tio.Open_Conf > Can't create directory: " & Dir_File);
222              end if;
223           end if;
224     
225           -- Apply optional permissions
226           if Fls.Exists (Name) and not Empty (Permissions) then
227              Sys.Shell_Execute (+"chmod 0600 " & Name, SE_Result);
228              if SE_Result /= 0 then
229                 Log.Err  (+"Tio.Open_Conf > Can't apply permissions to: " & Name);
230               end if;
231           end if;
232     
233        end Open_Conf;
234     
235        procedure Open_Conf (Handle : in out File; Name : VString ;
236                            Wipe_Before_Process : Boolean := False ;
237                            Permissions : VString := +"") is
238        begin
239           Open_Conf (Handle, Vst.To_String (Name), Wipe_Before_Process, Permissions);
240        end Open_Conf;

294        function Read_File (File_Name : VString) return VString is
295           File_Handle : File;
296           Line_Buffer, Result_Buffer : VString := +"";
297        begin
298           if Fls.Exists (File_Name) then
299              Tio.Open_Read (File_Handle, File_Name);
300              while not (End_Of_File (File_Handle)) loop
301                 Get_Line (File_Handle, Line_Buffer);
302                 Result_Buffer := Result_Buffer & Line_Buffer & LF;
303              end loop;
304              Close (File_Handle);
305           else
306              Log.Err (+"Tio.Read_File > File does not exist: " & File_Name);
307           end if;
308           return Result_Buffer;
309        end Read_File;

312        procedure Write_File (File_Name : VString ; Content : VString ; Permissions : VString := +"") is
313           Index_Next : Integer := 1;
314           String_To_Process : VString := +"";
315           Dir_File : constant VString := Fls.Extract_Directory (File_Name);
316           File_Handle : File;
317           SE_Result : Integer := 0;
318        begin
319     
320           -- Preventive housekeeping
321           Fls.Delete_File (File_Name);
322     
323           -- Ensure that the full tree structure exists before creating file
324           if Fls.Create_Directory_Tree (Dir_File) then
325              Create (File_Handle, File_Name);
326     
327              -- Apply optional permissions
328              if Fls.Exists (File_Name) and not Empty (Permissions) then
329                 Sys.Shell_Execute (+"chmod 0600 " & File_Name, SE_Result);
330                 if SE_Result /= 0 then
331                    Log.Err  (+"Tio.Open_Conf > Can't apply permissions to: " & File_Name);
332                 end if;
333              end if;
334     
335              -- Adding trailing LF if missing
336              String_To_Process := Content & (if Slice (Content, Length (Content)) = LF then +"" else To_VString (LF));
337              while True loop
338                 Index_Next := Index (String_To_Process, LF);
339                 Put_Line (File_Handle, Slice (String_To_Process, 1, Index_Next - 1));
340                 String_To_Process := Slice (String_To_Process, Index_Next + 1, Length (String_To_Process));
341                 if Length (String_To_Process) <= 1 then -- Last field reached
342                    exit;
343                 end if;
344              end loop;
345     
346              Close (File_Handle);
347     
348           else
349              Log.Err ("Tio.Write_File > Can't create directory: " & Dir_File);
350           end if;
351     
352        end Write_File;
