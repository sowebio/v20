58        Common_Compiler_Options := (
59         -- General 
60         "-gnatW8",           --  Both brackets and UTF-8 encodings will be recognized (1)
61         -- Warnings & Errors
62         "-gnatU",            --  Enable unique tag for error messages
63         "-gnatf",            --  Full errors. Verbose details, all undefined references
64         "-gnatq",            --  Don't quit, try semantics, even if parse errors
65         "-gnatQ",            --  Don't quit, write ali/tree file even if compile errors
66         "-gnatVaep",         --  Enable selected validity checking mode (2)   
67         "-gnatwDH.Y",      --  Enable selected warning modes (3)
68         -- "-Wall",          --  Enable most warning messages
69         -- Style
70         "-gnatyaefhkM160npr" --  Enable selected style checks (4)
71        );

73        Fast_Compiler_Options := (
74         "-O2",
75         "-gnatpn", 
76         "-fipa-cp-clone", "-fgcse-after-reload", 
77         "-funroll-loops", "-fpeel-loops", "-funswitch-loops", 
78         "-ftracer", "-fweb", "-ftree-vectorize",
79         "-frename-registers", "-ffunction-sections",
80         "-g"

83        Debug_Compiler_Options := (
84         "-gnata",            --  Assertions enabled
85         "-gnato",            --  Enable overflow checking in STRICT mode
86         "-gnateE",           --  Generate extra information in exception messages 
87         "-gnateF",           --  Check overflow on predefined Float types
88         "-gnatVa",           --  Enable all validity checking options
89         "-fstack-check",

96        Style_Compiler_Options := (
97        "-gnatg"              --  RTS Style (6)
98        );
99     
100        --  (1)
101        --  https://gcc.gnu.org/onlinedocs/gcc-4.8.5/gnat_ugn_unw/Character-Set-Control.html
102        --  https://gcc.gnu.org/onlinedocs/gcc-4.8.5/gnat_ugn_unw/Wide-Character-Encodings.html#Wide-Character-Encodings
103        --  (2) 
104        --  a turn on all validity checking options
105        --  e turn on checking for elementary components
106        --  p turn on checking for parameters
107        --  (3)
108        --  .e turn on every optional info/warning (no exceptions)
109        --  D  turn off warnings for implicit dereference (default)
110        --  H  turn off warnings for hiding declarations (default)
111        --  .Y turn off info messages for why pkg body needed (default)
112        --  (4)
113        --  a  check attribute casing
114        --  e  check end/exit labels present
115        --  f  check no form feeds/vertical tabs in source
116        --  h  no horizontal tabs in source
117        --  k  check casing rules for keywords
118        --  Mn check line length <= n characters
119        --  n  check casing of package Standard identifiers
120        --  p  check pragma casing
121        --  r  check casing for identifier references
122        --  (5)
123        --  Options starting with -g, -f, -m, -O, -W, or --param are automatically passed on to the various sub-processes 
124        --  invoked by gcc.  In order to pass  other options on to these processes the -W<letter> options must be used.
125        --  (6) All warnings and style messages are treated as errors. -gnatg implies -gnatw.ge and -gnatyg so that all
126        --  standard warnings and all standard style options are turned on. All warnings and style messages are treated
127        --  as errors.'

129        --  gnatmake options
130        package Compiler is
131           case v20_Build is
132           when "Style"   => 
133              for Default_Switches ("ada") use Common_Compiler_Options & Debug_Compiler_Options & Style_Compiler_Options;
134           when "Debug" => 
135              for Default_Switches ("ada") use Common_Compiler_Options & Debug_Compiler_Options;
136              for Switches ("s-memory.adb") use ("-gnatg");    
137           when "Fast"  => 
138              for Default_Switches ("ada") use Common_Compiler_Options & Fast_Compiler_Options;
139              for Switches ("s-memory.adb") use ("-gnatg");      
140           when "Small" =>
141              for Default_Switches ("ada") use Common_Compiler_Options & "-Os";
142              for Switches ("s-memory.adb") use ("-gnatg");                  
143           end case;
144        end Compiler;
