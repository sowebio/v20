41        procedure Exception_Handling
42                      (Exception_Hook : Ada.Exceptions.Exception_Occurrence) is
43     
44           Exception_Handle : Tio.File;
45           Exception_File_Name : constant VString := Prg.Start_Dir & "/" &
46                                                    Prg.Name & ".err";
47     
48           Trace_Output : VString := To_VString (Ada.Exceptions.Exception_Information (Exception_Hook));
49           Trace_Lines : constant Natural := Field_Count (Trace_Output, LF);
50           Current_Line : VString;
51           Current_Address : VString;
52           Number_Position, First_Space_Position : Natural;
53           Output_Processed : VString := +"";
54     
55           Trace_without_Lines : Boolean := False;
56     
57           SE_Result : Integer := 0;
58           SE_Output : VString := +"";
59     
60        begin
61     
62           Tio.Line;
63           Tio.Put_Line (Title_Max_Length * "-");
64           Tio.Line;
65           Tio.Put_Line ("Exception time         : " & Prg.Time_Stamp);
66           Tio.Put_Line ("Program uptime         : " & Prg.Duration_Stamp (Prg.Start_Time));
67           Tio.Put_Line ("Program build DT stamp : " & v20.Get_Build);
68           Tio.Put_Line ("Program name & version : " & Prg.Get_Version);
69           Tio.Put_Line ("Library name & version : " & v20.Get_Version);
70           Tio.Put_Line ("Start directory        : " & Prg.Start_Dir);
71           Tio.Put_Line ("Home directory         : " & Sys.Get_Home);
72           Tio.Put_Line ("Ada mem. alloc. (bytes): " & Sys.Get_Alloc_Ada);
73           Tio.Put_Line ("All mem. alloc. (bytes): " & Sys.Get_Alloc_All);
74     
75           --  In the GNAT Pro and CE versions AdaCore has integrated the GNU/GCC
76           --  utility addr2line into the GNAT runtime. It is called to display the
77           --  subroutine names and line numbers of the call stack.
78           --  This feature is missing in the GNAT FSF version. This is a recurring
79           --  complaint on the net, some people offer workarounds, but none of them
80           --  work here (Linux, GNAT FSF 11).
81           --  Since the CE versions are terminated, there were two solutions to
82           --  restore this useful feature; the clean one (build a full FSF 12
83           --  suite as GNAT FSF 12 gracefully returns line numbers - but lack of
84           --  time) and the dirty (faster).
85     
86           --  When handling an exception, we search if the addr2line utility exists
87           --  on the system and then, for each line indicating a "???", we run, on
88           --  the executable itself, a command addr2line -e <executable name>
89           --  <address> and analyze the return. If the latter is valid (example:
90           --  /home/sr/.../debug/gnx-utl_sv.adb:1174, as opposed to "addr2line:
91           --  DWARF error: can't find .debug_ranges section." and/or "??:?",
92           --  ??:0"), the function replaces "???" by "gnx-utl_sv.adb:1174". If the
93           --  program was compiled with GNAT Pro or CE, this is transparent and if
94           --  add2line was not found, the "???" are left as they are.
95     
96           if (Trace_Lines > 0) and Sys.Is_Command ("addr2line") then
97     
98              for I in 1 .. Trace_Lines loop
99                 Current_Line := Field_By_Index (Trace_Output, I, LF);
100                 Number_Position := Index (Current_Line, "???");
101                 First_Space_Position := Index (Current_Line, SP);
102     
103                 if (Number_Position > 2) and (First_Space_Position > 2) then
104                    Current_Address := Slice (Current_Line, 1, Index (Current_Line, SP) - 1);
105                    Sys.Shell_Execute ("addr2line -e " & Prg.Name & " " & Current_Address, SE_Result, SE_Output);
106                    if (Index (SE_Output, "/") > 0) then
107                       Output_Processed := Output_Processed & Slice (Current_Line, 1, Number_Position - 1) & Tail_After_Match (SE_Output, "/") & LF;
108                    else
109                       Output_Processed := Output_Processed & Current_Line & LF;
110                    end if;
111                    Trace_without_Lines := True;
112                 else
113                    Output_Processed := Output_Processed & Current_Line & LF;
114                 end if;
115              end loop;
116     
117              if Trace_without_Lines then
118                 Trace_Output := Output_Processed;
119              end if;
120     
121              Tio.Line;
122              Tio.Put_Line (Trace_Output);
123              Tio.Put_Line (Title_Max_Length * "-");
124              Tio.Line;
125     
126              if Ada.Directories.Exists (To_String (Exception_File_Name)) then
127                 Tio.Append (Exception_Handle, Exception_File_Name);
128              else
129                 Tio.Create (Exception_Handle, Exception_File_Name);
130              end if;
131     
132              if Tio.Is_Open (Exception_Handle) then
133                 Tio.Put_Line (Exception_Handle, Title_Max_Length * "-");
134                 Tio.Line (Exception_Handle);
135                 Tio.Put_Line (Exception_Handle, "Exception time         : " & Prg.Time_Stamp);
136                 Tio.Put_Line (Exception_Handle, "Program uptime         : " & Prg.Duration_Stamp (Prg.Start_Time));
137                 Tio.Put_Line (Exception_Handle, "Program build DT stamp : " & v20.Get_Build);
138                 Tio.Put_Line (Exception_Handle, "Program name & version : " & Prg.Get_Version);
139                 Tio.Put_Line (Exception_Handle, "Library name & version : " & v20.Get_Version);
140                 Tio.Put_Line (Exception_Handle, "Start directory        : " & Prg.Start_Dir);
141                 Tio.Put_Line (Exception_Handle, "Home directory         : " & Sys.Get_Home);
142                 Tio.Put_Line (Exception_Handle, "Ada mem. alloc. (bytes): " & Sys.Get_Alloc_Ada);
143                 Tio.Put_Line (Exception_Handle, "All mem. alloc. (bytes): " & Sys.Get_Alloc_All);
144                 Tio.Line (Exception_Handle);
145                 Tio.Put_Line (Exception_Handle, Trace_Output);
146                 Tio.Put_Line (Exception_Handle, Title_Max_Length * "-");
147                 --Tio.Line (Exception_Handle);
148                 Tio.Close (Exception_Handle);
149              end if;
150     
151           end if;
152     
153           Tio.Cursor_On;
154           Prg.Set_Exit_Status (9);
155     
156        end Exception_Handling;

158        function Get_Version return ASU.Unbounded_String is
159        begin
160           return (ASU.To_Unbounded_String (Name & " v" &
161                   Ada.Strings.Fixed.Trim (Natural'Image (Version_Major),
162                   Ada.Strings.Left) & "." &
163                   Ada.Strings.Fixed.Trim (Natural'Image (Version_Minor),
164                 Ada.Strings.Left)));
165        end Get_Version;
166     
167        function Get_Build return ASU.Unbounded_String is
168        begin
169           return (ASU.To_Unbounded_String ("build " &
170                   GNAT.Source_Info.Compilation_ISO_Date & " " &
171                   GNAT.Source_Info.Compilation_Time));
172        end Get_Build;
173     
174        function Get_Log_Dir return ASU.Unbounded_String is
175        begin
176          return ASU.To_Unbounded_String (Log_Dir);
177        end Get_Log_Dir;
178     
179        function Get_Tmp_Dir return ASU.Unbounded_String is
180        begin
181          return ASU.To_Unbounded_String (Tmp_Dir);
182        end Get_Tmp_Dir;
