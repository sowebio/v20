116        --  Form parameter:
117        --      The allowed values for preserve= are:
118        --        no_attributes:  Do not try to preserve any file attributes. This
119        --                        is the default if no preserve= is found in Form.
120        --        all_attributes: Try to preserve all file attributes (timestamps,
121        --                        access rights).
122        --        timestamps:     Preserve the timestamp of the copied file, but not
123        --                        the other file attributes.
124        --
125        --      The allowed values for mode= are:
126        --        copy:           Only copy if the destination file does not already
127        --                        exist. If it already exists, Copy_File will fail.
128        --        overwrite:      Copy the file in all cases. Overwrite an already
129        --                        existing destination file. This is the default if
130        --                        no mode= is found in Form.
131        --        append:         Append the original file to the destination file.
132        --                        If the destination file does not exist, the
133        --                        destination file is a copy of the source file.
134        --                        When mode=append, the field preserve=, if it
135        --                        exists, is not taken into account.
136     
137        Copy_Form : constant String := "preserve=all_attributes,mode=overwrite";
138        --  Copy_Form : constant String := "preserve=no_attributes,mode=overwrite";
