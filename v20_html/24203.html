37        function Current_Time_Seconds return Natural is
38        begin
39           return Integer (AC.Clock - ACF.Time_Of (1970, 1, 1, 0.0));
40        end Current_Time_Seconds;

42        function Duration_Stamp (Time : Ada.Calendar.Time) return VString is
43           Day_Secs : Natural;
44        begin
45           Day_Secs := (if AC.Seconds (AC.Clock) < AC.Seconds (Time) then 86400 else 0) +
46                           Integer (AC.Seconds (AC.Clock) - AC.Seconds (Time));
47           --  Unlimited hours counter for long uptimes
48           return To_VString  ((Day_Secs / 3600)) & "h" &
49                  Time_Format ((Day_Secs / 60) mod 60) & "m" &
50                  Time_Format  (Day_Secs mod 60) & "s";
51        end Duration_Stamp;
52     
53        function Duration_Stamp_Seconds (Time : Ada.Calendar.Time) return Natural is
54        begin
55           return (if AC.Seconds (AC.Clock) < AC.Seconds (Time) then 86400 else 0) +
56                           Integer (AC.Seconds (AC.Clock) - AC.Seconds (Time));
57        end Duration_Stamp_Seconds;
58     
59        function Duration_Stamp_Time (Time_Seconds : Integer) return VString is
60        begin
61           --  Unlimited hours counter for long uptimes
62           return To_VString  ((Time_Seconds / 3600)) & "h" &
63                  Time_Format ((Time_Seconds / 60) mod 60) & "m" &
64                  Time_Format  (Time_Seconds mod 60) & "s";
65        end Duration_Stamp_Time;

67        function Generate_Password return VString is
68           --  Sowebio password generation with 64 chars:
69           --    ([A-Z] + [a-z] + [0-9] + '_' + '-')
70           --  Search space size:
71           --    > 1,26 x 10^25
72           --  Space exploration time:
73           --    40000 centuries @ 100 billion tests per second.
74           --  Command line with standard tools:
75           --    < /dev/urandom tr -dc _A-Z-a-z-0-9 | head -c${1:-14};echo;
76           --  Generates 14 chars long passwords like:
77           --    x2U5hhIKX7IJt6
78           SE_Result : Integer := 0;
79           SE_Output : VString := +"";
80        begin
81           Sys.Shell_Execute ("< /dev/urandom tr -dc _A-Z-a-z-0-9 | head -c${1:-14};echo;", SE_Result, SE_Output);
82           if (SE_Result /= 0) then
83              Log.Err ("v20.Prg.Generate_Password > Password generation command failed");
84           end if;
85           return SE_Output;
86        end Generate_Password;

89        begin

94               Ada.Strings.Left));

107        function Is_User_Not_Root return Boolean is
108        begin
109           --  Determining if the current user is root is not so simple. Indeed, USER
110           --  and USERNAME can contain a "root" user who is not named "root" (juste
111           --  declare another user with a UID of 0). Two solutions are possible:
112           --  - Use the HOME environment variable, which allows to determine if the
113           --    user's home is "root".
114           --  - Use the command "whoami" which will return "root", even if the user
115           --  "root" has another name.
116                   return (not (Sys.Get_Env ("HOME") = +"/root"));
117        end Is_User_Not_Root;

119        function Name return VString is
120           Result : VString := Tail_After_Match (Command, "/");
121        begin
122           if (Length (Result) = 0) then
123              Result := Command;
124           end if;
125          return Result;
126        end Name;
127     
128        function Path return String is
129           Result_Code : Natural := 0;
130           Result_String : VString := +"";
131        begin
132           Sys.Shell_Execute ("pwd", Result_Code, Result_String);
133           -- pwd at root returns /
134           -- pwn elsewhere returns a path without a trailing slash
135           if (Slice (Result_String, Length (Result_String), Length (Result_String)) /= +"/") then
136              Result_String := Result_String & "/";
137           end if;
138           return To_String (Result_String);
139        end Path;

154        function Time_Format (Input_To_Format : Integer) return VString is
155           Two_Digits_Output : VString;
156        begin
157           Two_Digits_Output := To_VString (Input_To_Format);
158           if Input_To_Format < 10 then
159              Two_Digits_Output := +"0" & Two_Digits_Output;
160           end if;
161           return Two_Digits_Output;
162        end Time_Format;
163     
164        function Time_Stamp return VString is
165        begin
166           return To_VString (GCT.Image (AC.Clock, "%Y%m%d-%H%M%S"));
167        end Time_Stamp;
