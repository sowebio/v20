33        procedure Raise_Exception is
34           v20_Exception_Test : exception;
35        begin
36           raise v20_Exception_Test;
37        end Raise_Exception;
38     
39        procedure Exception_Handling
40                      (Exception_Hook : Ada.Exceptions.Exception_Occurrence) is
41     
42           Exception_Handle : Tio.File;
43           Exception_File_Name : constant String := Prg.Start_Dir & "/" &
44                                                    Prg.Name & ".err";
45        begin
46     
47           Tio.Put_Line (Line_Max_Length * "-");
48           Tio.Line;
49           Tio.Put_Line ("Exception time        : " & Prg.Time_Stamp);
50           Tio.Put_Line ("Program uptime        : " &
51                          Prg.Duration_Stamp (Prg.Start_Time));
52           Tio.Put_Line ("Program name & version: " & Prg.Get_Version);
53           Tio.Put_Line ("Library name & version: " & v20.Get_Version);
54           Tio.Put_Line ("Start directory       : " & Prg.Start_Dir);
55           Tio.Put_Line ("Home directory        : " & Sys.Get_Home);
56           Tio.Put_Line ("Ada memory allocations: " & Sys.Get_Alloc_Ada);
57           Tio.Put_Line ("All memory allocations: " & Sys.Get_Alloc_All);
58     
59           Tio.Line;
60           Tio.Put_Line (Ada.Exceptions.Exception_Information (Exception_Hook));
61           Tio.Put_Line (Line_Max_Length * "-");
62           Tio.Line;
63     
64           if Ada.Directories.Exists (Exception_File_Name) then
65              Tio.Append (Exception_Handle, Exception_File_Name);
66           else
67              Tio.Create (Exception_Handle, Exception_File_Name);
68           end if;
69     
70           if Tio.Is_Open (Exception_Handle) then
71              Tio.Put_Line (Exception_Handle, Line_Max_Length * "-");
72              Tio.Line (Exception_Handle);
73              Tio.Put_Line (Exception_Handle, "Exception time        : " &
74                                               Prg.Time_Stamp);
75              Tio.Put_Line (Exception_Handle, "Program uptime        : " &
76                                               Prg.Duration_Stamp (Prg.Start_Time));
77              Tio.Put_Line (Exception_Handle, "Program name & version: " &
78                                               Prg.Get_Version);
79              Tio.Put_Line (Exception_Handle, "Library name & version: " &
80                                               v20.Get_Version);
81              Tio.Put_Line (Exception_Handle, "Start directory       : " &
82                                               Prg.Start_Dir);
83              Tio.Put_Line (Exception_Handle, "Home directory        : " &
84                                               Sys.Get_Home);
85              Tio.Put_Line (Exception_Handle, "Ada memory allocations: " &
86                                               Sys.Get_Alloc_Ada);
87              Tio.Put_Line (Exception_Handle, "All memory allocations: " &
88                                               Sys.Get_Alloc_All);
89     
90              Tio.Line (Exception_Handle);
91              Tio.Put_Line (Exception_Handle,
92                             Ada.Exceptions.Exception_Information (Exception_Hook));
93              Tio.Put_Line (Exception_Handle, Line_Max_Length * "-");
94              Tio.Line (Exception_Handle);
95              Tio.Close (Exception_Handle);
96           end if;
97     
98           Prg.Set_Exit_Status (128);
99     
100        end Exception_Handling;
101     
102        function Get_Version return String is
103        begin
104           return (Name & " v" &
105                   Ada.Strings.Fixed.Trim (Natural'Image (Version_Major),
106                   Ada.Strings.Left) & "." &
107                   Ada.Strings.Fixed.Trim (Natural'Image (Version_Minor),
108                 Ada.Strings.Left));
109        end Get_Version;
