33        function Table_Write (Line : VString) return Boolean is
34           Result : Boolean := True;
35        begin
36           if Cfg_Last = Table_Max then
37              Log.Err ("v20.Cfg.Table_Write - Cfg.Table_Write > too large to load: " & Cfg_File_Read);
38              Cfg_Last := 0;
39              Result := False;
40           else
41              Cfg_Last := Cfg_Last + 1;
42              Cfg_Table (Cfg_Last) := Line;
43           end if;
44           return Result;
45        end Table_Write;
46     
47        ----------------------------------------------------------------------------
48        function Cfg_Read return Boolean is
49           Line_Buffer : VString := +"";
50           Result : Boolean := True;
51        begin
52           Cfg_Last := 0;
53           Tio.Reset (Handle_Read);
54           while (not Tio.End_Of_File (Handle_Read)) and Result loop
55              Tio.Get_Line (Handle_Read, Line_Buffer);
56              Result := Table_Write (Line_Buffer);
57           end loop;
58           return Result;
59        end Cfg_Read;

62        function Check_Parameters (Section : String;
63                                   Parameter : String;
64                                   Value : String)
65                                   return Boolean is
66           String_To_Test : constant VString :=
67                                          To_VString (Section & Parameter & Value);
68        begin
69           return (Index (String_To_Test, Cfg_Open_Section) +
70                   Index (String_To_Test, Cfg_Close_Section) +
71                   Index (String_To_Test, Cfg_Assignment) +
72                   Index (String_To_Test, Cfg_Comment) = 0);
73        end Check_Parameters;

75        procedure Cfg_Search (Section : String; Parameter : String) is
76           Pos_Open_Section : Natural := 0;
77           Pos_Close_Section : Natural := 0;
78           Pos_Assignment : Natural := 0;
79        begin
80           Cfg_Section := 0;
81           Cfg_Parameter := 0;
82           for I in 1 .. Cfg_Last loop
83              --  Section found, search for parameter
84              if Cfg_Section > 0 then
85                 --  Section break, parameter not found
86                 if Index (Cfg_Table (I), Cfg_Open_Section) > 0 and
87                    Index (Cfg_Table (I), Cfg_Close_Section) > 0
88                 then
89                    exit;
90                 else
91                    Pos_Assignment := Index (Trim_Left (Cfg_Table (I)),
92                                             Cfg_Assignment);
93                    --  At least a parameter name of one char
94                    if Pos_Assignment > 2 then
95                       --  Parameter found
96                       if Parameter = Trim_Both (Slice (Cfg_Table (I), 1,
97                                                       Pos_Assignment - 1))
98                       then
99                          Cfg_Parameter := I;
100                          exit;
101                       end if;
102                    end if;
103                 end if;
104                 --  Search for section
105              else
106                 Pos_Open_Section := Index (Cfg_Table (I), Cfg_Open_Section);
107                 if Pos_Open_Section > 0 then
108                    Pos_Close_Section := Index (Cfg_Table (I), Cfg_Close_Section);
109                    --  At least a section name of one char
110                    if Pos_Close_Section > 0 and
111                       (Pos_Close_Section - Pos_Open_Section > 1)
112                    then
113                       if Section = Trim_Both (Slice (Cfg_Table (I),
114                                  Pos_Open_Section + 1, Pos_Close_Section - 1))
115                       then
116                          Cfg_Section := I;
117                       end if;
118                    end if;
119                 end if;
120              end if;
121           end loop;
122        end Cfg_Search;

125        function Cfg_Write (Section : String := "";
126                            Parameter : String := "";
127                            Value : String := "";
128                            Trailing_Comment : VString := +"")
129                            return Boolean is
130           Result : Boolean := True;
131        begin
132     
133           Tio.Create (Handle_Write, Cfg_File_Write);
134           if Tio.Is_Open (Handle_Write) then
135              for I in 1 .. Cfg_Last loop
136     
137                 --  Create Parameter
138                 if Index (Cfg_Table (I), Cfg_Command_Add) > 0 then
139                    if Index (Cfg_Table (I), Cfg_Comment) > 0 then
140                       Cfg_Table (I) := Cfg_Open_Section & Section &
141                                        Cfg_Close_Section &  " " &
142                       Slice (Cfg_Table (I), Index (Cfg_Table (I),
143                              Cfg_Comment),
144                              Length (Cfg_Table (I)));
145                    else
146                       Cfg_Table (I) := Cfg_Open_Section & Section &
147                                        Cfg_Close_Section;
148                    end if;
149                    Tio.Put_Line (Handle_Write, Cfg_Table (I));
150                    Tio.Put_Line (Handle_Write, Parameter & " "  &
151                                                Cfg_Assignment & " " &
152                                                Value &
153                                                Trailing_Comment);
154                 --  Delete Line
155                 elsif Index (Cfg_Table (I), Cfg_Command_Delete) > 0 then
156                    null;
157     
158                 --  Copy Line
159                 else
160                    Tio.Put_Line (Handle_Write, Cfg_Table (I));
161                 end if;
162     
163              end loop;
164     
165              --  Replacing old cfg file by the updated new one
166              Tio.Close (Handle_Write);
167              Tio.Close (Handle_Read);
168              Fls.Delete_File (To_String (Cfg_File_Read));
169              Fls.Rename (Cfg_File_Write, To_String (Cfg_File_Read));
170              Tio.Open_Read (Handle_Read, To_String (Cfg_File_Read));
171     
172              if Tio.Is_Open (Handle_Read) then
173                 Result := Cfg_Read;
174              else
175                 Log.Err ("v20.Cfg.Cfg_Write - Can't open new " & Cfg_File_Read);
176                 Result := False;
177              end if;
178           else
179              Log.Err ("v20.Cfg.Cfg_Write - Can't create " & Cfg_File_Write);
180              Result := False;
181           end if;
182           return Result;
183        end Cfg_Write;

186        procedure Set (Section : String;
187                       Parameter : String;
188                       Value : String;
189                       Comment : String := "") is
190     
191           Trailing_Comment : VString := +"";
192        begin
193           --  Check open/close section, assignment and comment reserved characters
194           --  in Section, Parameter and Value
195           if Check_Parameters (Section, Parameter, Value) then
196     
197              Cfg_Search (Section, Parameter);
198     
199              if Comment /= "" then
200                 Trailing_Comment := To_VString ("  # " & Comment);
201              end if;
202     
203              --  Update parameter
204              if Cfg_Section > 0 and Cfg_Parameter > 0 then
205     
206                 --  Preserve parameter comment when updating
207                 if Index (Cfg_Table (Cfg_Parameter), Cfg_Comment) > 0 then
208                    Cfg_Table (Cfg_Parameter) := Parameter & " " & Cfg_Assignment &
209                                                 " " & Value & " " &
210                    Slice (Cfg_Table (Cfg_Parameter),
211                           Index (Cfg_Table (Cfg_Parameter), Cfg_Comment),
212                           Length (Cfg_Table (Cfg_Parameter)));
213                 else
214                    Cfg_Table (Cfg_Parameter) := Parameter & " " &
215                                                 Cfg_Assignment & " " &
216                                                 Value &
217                                                 Trailing_Comment;
218                 end if;
219     
220              --  Create Parameter => tag the section for further just in time
221              --  write in file
222              elsif Cfg_Section > 0 then
223     
224                 --  Preserve section comment when tagging
225                 if Index (Cfg_Table (Cfg_Section), Cfg_Comment) > 0 then
226                    Cfg_Table (Cfg_Section) := Cfg_Command_Add & " " &
227                           Slice (Cfg_Table (Cfg_Section),
228                           Index (Cfg_Table (Cfg_Section), Cfg_Comment),
229                           Length (Cfg_Table (Cfg_Section)));
230                 else
231                    Cfg_Table (Cfg_Section) := Cfg_Command_Add;
232                 end if;
233     
234                 --  Create Section and Parameter
235              else
236                 if Table_Max >= Cfg_Last + 2 then
237                    Cfg_Table (Cfg_Last + 1) := Cfg_Open_Section &
238                                                Section &
239                                                Cfg_Close_Section;
240                    Cfg_Table (Cfg_Last + 2) := Parameter & " " &
241                                                Cfg_Assignment & " " &
242                                                Value &
243                                                Trailing_Comment;
244                    Cfg_Last := Cfg_Last + 2;
245                 else
246                    Log.Err ("v20.Cfg.Set - Too large: " & Cfg_File_Read);
247                    Log.Err ("v20.Cfg.Set - For further writes to: " &
248                                                           Cfg_File_Write);
249                 end if;
250              end if;
251     
252              if not Cfg_Write (Section, Parameter, Value, Trailing_Comment) then
253                 Log.Err ("v20.Cfg.Set - Cfg_Write operation failed");
254              end if;
255           else
256              Log.Err (
257                 "v20.Cfg.Set - Invalid character in Section, Parameter or Value");
258           end if;
259        end Set;

262        function Get (Section : String; Parameter : String) return String is
263           Result : VString := +"";
264        begin
265     
266           Cfg_Search (Section, Parameter);
267     
268           --  Get parameter
269           if Cfg_Section > 0 and Cfg_Parameter > 0 then
270              Result := Cfg_Table (Cfg_Parameter);
271              --  Suppress comment
272              if Index (Result, Cfg_Comment) > 0 then
273                 Result := Slice (Result, 1, Index (Result, Cfg_Comment) - 1);
274              end if;
275              Result := Trim_Both (Slice (Result,
276                                          Index (Result, Cfg_Assignment) + 1,
277                                          Length (Result)));
278           end if;
279     
280           return To_String (Result);
281     
282        end Get;

284        procedure Delete (Section : String; Parameter : String) is
285           Count_Parameters : Natural := 0;
286        begin
287     
288           Cfg_Search (Section, Parameter);
289     
290           --  Get parameter
291           if Cfg_Section > 0 and Cfg_Parameter > 0 then
292              Cfg_Table (Cfg_Parameter) := Cfg_Command_Delete;
293     
294              --  Does Section owns at least one parameter ?
295              for I in (Cfg_Section + 1) .. Cfg_Last loop
296                 --  Section break
297                 if Index (Cfg_Table (I), Cfg_Open_Section) > 0 and
298                    Index (Cfg_Table (I), Cfg_Close_Section) > 0
299                 then
300                    exit;
301                 end if;
302                 --  Not the deleted parameter, not and empty line nor a
303                 --  commented line
304                 if (
305                    (I /= Cfg_Parameter) or
306                    (Index (Cfg_Table (I), Cfg_Assignment) > 0)) and
307                    not (Index (Cfg_Table (I), Cfg_Comment) > 0)
308                 then
309                    Count_Parameters := Count_Parameters + 1;
310                 end if;
311              end loop;
312              --  Delete section too
313              if Count_Parameters = 0 then
314                 Cfg_Table (Cfg_Section) := Cfg_Open_Section &
315                                            Cfg_Open_Section & "D";
316              end if;
317     
318              if not Cfg_Write then
319                 Log.Err ("v20.Cfg.Delete - Cfg_Write operation failed");
320              end if;
321           end if;
322        end Delete;

325        procedure Comment (Text : String) is
326           Dummy : Boolean;
327        begin
328           if Table_Write (To_VString ("# " & Text)) then
329              Dummy := Cfg_Write;
330           end if;
331        end Comment;

334        function Open (Cfg_File_Read_In : String := "") return Boolean is
335           Result : Boolean := False;
336        begin
337           --  Custom name
338           if Cfg_File_Read_In /= "" then
339              Cfg_File_Read := To_VString (Cfg_File_Read_In);
340           end if;
341           --  Create if non-existent
342           if not Ada.Directories.Exists (To_String (Cfg_File_Read)) then
343              Tio.Create (Handle_Read, To_String (Cfg_File_Read));
344              if Tio.Is_Open (Handle_Read) then
345                 Close;
346              else
347                 Log.Err ("v20.Cfg.Open - Can't create Cnf file");
348              end if;
349           end if;
350           --  Open in read mode
351           Tio.Open_Read (Handle_Read, To_String (Cfg_File_Read));
352           if Tio.Is_Open (Handle_Read) then
353              Result :=  Cfg_Read;
354           end if;
355           return Result;
356        end Open;

359        procedure Close is
360        begin
361           if Tio.Is_Open (Handle_Read) then
362              Tio.Close (Handle_Read);
363           else
364              Log.Err ("v20.Cfg.close - Should be opened before closing: " &
365                        Cfg_File_Read);
366           end if;
367        end Close;
