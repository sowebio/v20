28     with System;  use System;
29     
30     package body Object.Handle is
31     
32        procedure Adjust (Reference : in out Handle) is
33        begin
34           if Reference.Ptr /= null then
35              Increment_Count (Reference.Ptr.all);
36           end if;
37        end Adjust;
38     
39        procedure Finalize (Reference : in out Handle) is
40        begin
41           Release (Reference.Ptr);
42           Reference.Ptr := null;
43        end Finalize;
44     
45        procedure Invalidate (Reference : in out Handle) is
46        begin
47           if Reference.Ptr /= null then
48              Release (Reference.Ptr);
49              Reference.Ptr := null;
50           end if;
51        end Invalidate;
52     
53        function Is_Valid (Reference : Handle) return Boolean is
54        begin
55           return Reference.Ptr /= null;
56        end Is_Valid;
57     
58        function Ptr (Reference : Handle) return Object_Type_Ptr is
59        begin
60           return Reference.Ptr;
61        end Ptr;
62     
63        function Ref (Thing : Object_Type_Ptr) return Handle is
64        begin
65           if Thing /= null then
66              Increment_Count (Thing.all);
67           end if;
68           return Handle'(Ada.Finalization.Controlled with Thing);
69        end Ref;
70     
71        procedure Release (Ptr : in out Object_Type_Ptr) is
72        begin
73           if Ptr /= null then
74              declare
75                 --
76                 -- We  have  to  use  an  unchecked  access  here,   because
77                 -- otherwise   accessibility   check   might    fail    when
78                 -- Object_Type_Ptr is a local access type.
79                 --
80                 Pointer : Entity_Ptr := Ptr.all'Unchecked_Access;
81              begin
82                 Release (Pointer);
83                 if Pointer = null then
84                    Ptr := null;
85                 end if;
86              end;
87           end if;
88        end Release;
89     
90        procedure Set (Reference : in out Handle; Thing : Object_Type_Ptr) is
91        begin
92           if Reference.Ptr /= Thing then
93              if Reference.Ptr /= null then
94                 Release (Reference.Ptr);
95                 Reference.Ptr := null;
96              end if;
97              if Thing /= null then
98                 Increment_Count (Thing.all);
99                 Reference.Ptr := Thing;
100              end if;
101           end if;
102        end Set;
103     
104        function "<" (Left, Right : Handle) return Boolean is
105        begin
106           return Less (Left.Ptr.all, Right.Ptr.all);
107        end "<";
108     
109        function "<=" (Left, Right : Handle) return Boolean is
110        begin
111           return
112           (  Equal (Left.Ptr.all, Right.Ptr.all)
113           or else
114              Less (Left.Ptr.all, Right.Ptr.all)
115           );
116        end "<=";
117     
118        function "=" (Left, Right : Handle) return Boolean is
119        begin
120           if Left.Ptr = null then
121              if Right.Ptr = null then
122                 return True;
123              else
124                 return False;
125              end if;
126           else
127              if Right.Ptr = null then
128                 return False;
129              else
130                 return Equal (Left.Ptr.all, Right.Ptr.all);
131              end if;
132           end if;
133        end "=";
134     
135        function ">=" (Left, Right : Handle) return Boolean is
136        begin
137           return not Less (Left.Ptr.all, Right.Ptr.all);
138        end ">=";
139     
140        function ">" (Left, Right : Handle) return Boolean is
141        begin
142           return
143              not
144              (  Equal (Left.Ptr.all, Right.Ptr.all)
145              or else
146                 Less (Left.Ptr.all, Right.Ptr.all)
147              );
148        end ">";
149     
150        function "=" (Left : Handle; Right : access Object_Type'Class)
151           return Boolean is
152        begin
153           if Left.Ptr = null then
154              return Right = null;
155           elsif Right = null then
156              return False;
157           else
158              return Left.Ptr.all'Address = Right.all'Address;
159           end if;
160        end "=";
161     
162        function "=" (Left : access Object_Type'Class; Right : Handle)
163           return Boolean is
164        begin
165           if Right.Ptr = null then
166              return Left = null;
167           elsif Left = null then
168              return False;
169           else
170              return Right.Ptr.all'Address = Left.all'Address;
171           end if;
172        end "=";
173     
174     end Object.Handle;
