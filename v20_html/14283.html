64        --  Redirection constants
65        STD_OUT_REDIRECT  : constant String := " 1>/dev/null";

73        -- Force a true (without an apostrophe pair generated by 'Image attribute) 
74        -- character to string conversion by adding a trailing empty string.
75        HT   : constant String := Character'Val(9)  & ""; -- 09d 09h Tab
76        LF   : constant String := Character'Val(10) & ""; -- 10d 0Ah Line Feed
77        CR   : constant String := Character'Val(13) & ""; -- 13d 0Dh Carriage return
78        ESC  : constant String := Character'Val(27) & ""; -- 27d 1Bh Escape
79        DQ   : constant String := Character'Val(34) & ""; -- 34d 22h Double quote
80        CRLF : constant String := CR & LF;
81        
82        ND : constant String := "~"; -- 126d 7Eh Name/value delimiter
83        CD : constant String := "^"; --  94d 5Eh Column delimiter
84        RD : constant String := "\"; --  92d 5Ch Row delimiter
85        VD : constant String := ","; --  44d 2Ch Virgule (comma) delimiter
86        SD : constant String := ":"; --  58d 3Ah Colon delimiter
87        SP : constant String := " "; --  32d 20h Space
88     
89        -- ANSI colors (ISO 6429 strandard)
90        
91        CONSOLE_COLOR_GREEN  : constant String := ESC & "[1;32m";
92        CONSOLE_COLOR_RED    : constant String := ESC & "[1;31m";
93        CONSOLE_COLOR_YELLOW : constant String := ESC & "[1;33m";
94        CONSOLE_COLOR_RESET  : constant String := ESC & "[0m";
