32        procedure Put (Line_In : String;
33                       Line_Level : String;
34                       Title_On : Boolean := False) is
35           Line : VString := To_VString (Line_In);
36           Line_Disk : VString := Line;
37           Line_Task : VString := +"";
38           Ansi_Begin : VString := +"";
39           Ansi_End : VString := +"";
40        begin
41     
42           if Tio.Ansi then
43              if Line_Level = "DBG" then
44                 Ansi_Begin := To_VString (ASCII.ESC & "[1;33m");
45              elsif Line_Level = "ERR" then
46                 Ansi_Begin := To_VString (ASCII.ESC & "[1;31m");
47              end if;
48              Ansi_End := To_VString (ASCII.ESC & "[0m");
49           end if;
50     
51           --  Task length control
52           if Length (Task_State) > Task_Max_Length then
53              --                  these two numbers must be equals v     v
54              Line_Task := Slice (Task_State, 1, Task_Max_Length - 1) & (1 * "*");
55           elsif  Length (Task_State) < Task_Max_Length then
56              Line_Task :=  Task_State &
57                           (Task_Max_Length - Length (Task_State)) * " ";
58           else
59              Line_Task :=  Task_State;
60           end if;
61     
62           --  Header console and disk mode
63           if Header_On then
64     
65              --  Line Length control
66              if (Header_Length + Length (Line) + 1) > Line_Max_Length then
67                 --                                      these two numbers v
68                 Line := Slice (Line, 1, Line_Max_Length - Header_Length - 1) &
69                                                                         (1 * "*");
70                 --                                        must be equals ^                                                        
71              end if;
72     
73              if Title_On then
74                 if (Header_Length + Length (Line) + 1) < Line_Max_Length then
75                    Line := Line &
76                           (Line_Max_Length - Header_Length - Length (Line)) * "-";
77                 end if;
78              end if;
79              --  Console write with limited length line and ansi fancy
80              Tio.Put_Line (To_String (Prg.Time_Stamp & " - " &
81                                       Line_Task & " - " &
82                          Ansi_Begin & Line_Level & Ansi_End & " - " &
83                                       Line));
84           --  Free console mode
85           else
86              Tio.Put_Line (To_String (Line));
87           end if;
88           --  Disk write with unlimited length line
89           if Disk_On then
90              if Title_On then
91                 if (Header_Length + Length (Line_Disk) + 1) < Line_Max_Length then
92                    Line_Disk := Line_Disk &
93                     (Line_Max_Length - Header_Length - Length (Line_Disk)) * "-";
94                 end if;
95              end if;
96              Tio.Put_Line (Handle, To_String (Prg.Time_Stamp & " - " &
97                                               Line_Task & " - " &
98                                               Line_Level & " - " &
99                                               Line_Disk));
100           end if;
101     
102        end Put;

192        ----------------------------------------------------------------------------
193        procedure Set_Disk (Action : Boolean) is
194           Log_File_Name : constant String := To_String (Log_Dir_Store) & "/" &
195                                              Prg.Name & ".log";
196        begin
197           Disk_On := Action;
198           if Disk_On then
199              if Fls.Exists (Log_File_Name) then
200                 Tio.Append (Handle, Log_File_Name);
201              else
202                 Tio.Create (Handle, Log_File_Name);
203              end if;
204              if not Tio.Is_Open (Handle) then
205                 Disk_On := False;
206                 Err ("Log.Set_Disk > can't log on disk to: " & Log_File_Name);
207              end if;
208           else
209              if Tio.Is_Open (Handle) then
210                 Tio.Close (Handle);
211              end if;
212           end if;
213        end Set_Disk;
214
