39     with GNATCOLL.Memory;
40     
41     package body System.Memory is
42     
43        package M renames GNATCOLL.Memory;
44     
45        function Alloc (Size : size_t) return System.Address is
46        begin
47           return M.Alloc (M.size_t (Size));
48        end Alloc;
49     
50        procedure Free (Ptr : System.Address) renames M.Free;
51     
52        function Realloc
53           (Ptr  : System.Address;
54            Size : size_t)
55           return System.Address
56        is
57        begin
58           return M.Realloc (Ptr, M.size_t (Size));
59        end Realloc;
60     
61     ------------------------------------------------------------------------------
62     end System.Memory;
