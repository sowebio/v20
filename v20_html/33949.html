23     with Ada.Strings.Fixed;
24     
25     with v20.Sys;
26     
27     package body v20.Prg is
28     
29        --  Program related --------------------------------------------------------
30     
31        function Duration_Stamp (Start_Time : Ada.Calendar.Time) return VString is
32           Current_Time : constant Integer :=
33                              Integer (Ada.Calendar.Seconds (Ada.Calendar.Clock));
34           Day_Secs : constant Integer := Current_Time -
35                              Integer (Ada.Calendar.Seconds (Start_Time));
36        begin
37           --  Unlimited hours counter for long uptimes
38           return To_VString ((Day_Secs / 3600)) & "h" &
39                Time_Format ((Day_Secs / 60) mod 60) & "m" &
40                Time_Format (Day_Secs mod 60) & "s";
41        end Duration_Stamp;
42     
43        function Get_Version return String is
44        begin
45           return (Name & " v" &
46                   Ada.Strings.Fixed.Trim (Natural'Image (Version_Major),
47                   Ada.Strings.Left) & "." &
48                   Ada.Strings.Fixed.Trim (Natural'Image (Version_Minor),
49               Ada.Strings.Left));
50        end Get_Version;
51     
52        function Is_User_Not_Root return Boolean is
53        begin
54           return (not (Sys.Get_Env ("USER") = +"root"));
55        end Is_User_Not_Root;
56     
57        procedure Set_Exit_Status (Code : Natural) is
58        begin
59           Prg_Exit_Status := Prg_Exit_Status + Code;
60           Ada.Command_Line.Set_Exit_Status
61                                  (Ada.Command_Line.Exit_Status (Prg_Exit_Status));
62        end Set_Exit_Status;
63     
64        procedure Set_Version (Major : Natural; Minor : Natural) is
65        begin
66           Version_Major := Major;
67           Version_Minor := Minor;
68        end Set_Version;
69     
70        function Time_Format (Input_To_Format : Integer) return VString is
71           Two_Digits_Output : VString;
72        begin
73           Two_Digits_Output := To_VString (Input_To_Format);
74           if Input_To_Format < 10 then
75              Two_Digits_Output := +"0" & Two_Digits_Output;
76           end if;
77           return Two_Digits_Output;
78        end Time_Format;
79     
80        function Time_Stamp return VString is
81           Current_Time : constant Ada.Calendar.Time := Ada.Calendar.Clock;
82           Day_Secs : constant Integer :=
83                                    Integer (Ada.Calendar.Seconds (Current_Time));
84        begin
85           return To_VString (Ada.Calendar.Year (Current_Time)) &
86           Time_Format (Ada.Calendar.Month (Current_Time)) &
87           Time_Format (Ada.Calendar.Day (Current_Time)) &
88           " " &
89           Time_Format (Day_Secs / 3660) &
90           Time_Format ((Day_Secs / 60) mod 60) &
91           Time_Format (Day_Secs mod 60);
92        end Time_Stamp;
93
